cmake_minimum_required(VERSION 3.21)

# 1. Принудительно направляем поиск в MSYS2
list(APPEND CMAKE_PREFIX_PATH "C:/msys64/mingw64")

if(DEFINED ENV{VCPKG_ROOT})
    set(CMAKE_TOOLCHAIN_FILE "$ENV{VCPKG_ROOT}/scripts/buildsystems/vcpkg.cmake" CACHE STRING "")
endif()

project(SmartHomeLite LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 17)

# 2. Ищем пакеты
find_package(Crow CONFIG REQUIRED)
find_package(GTest CONFIG REQUIRED)
set(MSYS_LIB_DIR "C:/msys64/mingw64/lib")
set(MSYS_INC_DIR "C:/msys64/mingw64/include")

# Прямые пути к библиотекам MinGW (чтобы vcpkg не мешал)
set(LIB_PQXX "${MSYS_LIB_DIR}/libpqxx.dll.a")
set(LIB_PQ "${MSYS_LIB_DIR}/libpq.dll.a")
set(LIB_RABBIT "${MSYS_LIB_DIR}/librabbitmq.dll.a")
set(LIB_GTEST "${MSYS_LIB_DIR}/libgtest.dll.a")
set(LIB_GTEST_MAIN "${MSYS_LIB_DIR}/libgtest_main.dll.a")
set(LIB_GMOCK "${MSYS_LIB_DIR}/libgmock.dll.a")

add_definitions(-DPQXX_SHARED)
include_directories(BEFORE src ${MSYS_INC_DIR})

set(COMMON_SOURCES src/application/service/reading_service_impl.cpp)

# --- 3. СЕРВЕР ---
add_executable(smart_home_server src/main.cpp ${COMMON_SOURCES})
target_link_libraries(smart_home_server PRIVATE 
    Crow::Crow ${LIB_PQXX} ${LIB_PQ} ${LIB_RABBIT} 
    ws2_32 mswsock advapi32 secur32 crypt32
)

# --- 4. UNIT ТЕСТЫ ---
add_executable(smart_home_tests 
    tests/unit/test_sensor.cpp 
    tests/unit/test_register_handler.cpp 
    tests/unit/reading_service_test.cpp 
    ${COMMON_SOURCES}
)
# ДОБАВЛЕНО: ${LIB_PQXX} ${LIB_PQ}, так как ReadingServiceImpl теперь зависит от них
target_link_libraries(smart_home_tests PRIVATE 
    ${LIB_GMOCK} ${LIB_GTEST} ${LIB_GTEST_MAIN} 
    ${LIB_RABBIT} ${LIB_PQXX} ${LIB_PQ}
    ws2_32 advapi32 mswsock
)

# --- 5. INTEGRATION ТЕСТЫ ---
add_executable(smart_home_integration_tests tests/integration/test_postgres_integration.cpp ${COMMON_SOURCES})
target_link_libraries(smart_home_integration_tests PRIVATE 
    ${LIB_GTEST} ${LIB_GTEST_MAIN} 
    ${LIB_PQXX} ${LIB_PQ} ${LIB_RABBIT} 
    ws2_32 mswsock advapi32
)

include(GoogleTest)
gtest_discover_tests(smart_home_tests)
gtest_discover_tests(smart_home_integration_tests)
