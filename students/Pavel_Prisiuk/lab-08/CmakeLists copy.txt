cmake_minimum_required(VERSION 3.21)

# 1. vcpkg только для Crow и GTest (они совместимы)
if(DEFINED ENV{VCPKG_ROOT})
    set(CMAKE_TOOLCHAIN_FILE "$ENV{VCPKG_ROOT}/scripts/buildsystems/vcpkg.cmake" CACHE STRING "")
endif()

project(SmartHomeLite LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 17)

# 2. Ищем пакеты из vcpkg
find_package(Crow CONFIG REQUIRED)
find_package(GTest CONFIG REQUIRED)

# 3. ЖЕСТКИЙ ПОИСК PQXX В MSYS2 (Игнорируем vcpkg)
set(MSYS_PATH "C:/msys64/mingw64")
set(MSYS_INC "${MSYS_PATH}/include")
set(MSYS_LIB_DIR "${MSYS_PATH}/lib")

# Ищем конкретные файлы .dll.a
set(LIB_PQXX "${MSYS_LIB_DIR}/libpqxx.dll.a")
set(LIB_PQ "${MSYS_LIB_DIR}/libpq.dll.a")

# Глобальные инклуды: ПРИНУДИТЕЛЬНО MSYS2 ПЕРВЫМ
include_directories(BEFORE SYSTEM ${MSYS_INC})
include_directories(src)

# Глобальный макрос для динамической линковки
add_definitions(-DPQXX_SHARED)

set(COMMON_SOURCES src/application/service/reading_service_impl.cpp)

# --- 4. СЕРВЕР ---
add_executable(smart_home_server src/main.cpp ${COMMON_SOURCES})
target_link_libraries(smart_home_server PRIVATE 
    Crow::Crow 
    ${LIB_PQXX} 
    ${LIB_PQ}
    ws2_32 mswsock advapi32 secur32 crypt32 shell32
)

# --- 5. UNIT ТЕСТЫ ---
add_executable(smart_home_tests
    tests/unit/test_sensor.cpp
    tests/unit/test_register_handler.cpp
    tests/unit/reading_service_test.cpp
    ${COMMON_SOURCES}
)
target_link_libraries(smart_home_tests PRIVATE 
    GTest::gtest GTest::gtest_main GTest::gmock
    ${LIB_PQXX} ${LIB_PQ} # Тестам тоже нужен pqxx, так как он в хендлерах
    ws2_32 advapi32
)

# --- 6. INTEGRATION ТЕСТЫ ---
add_executable(smart_home_integration_tests
    tests/integration/test_postgres_integration.cpp
    ${COMMON_SOURCES}
)
target_link_libraries(smart_home_integration_tests PRIVATE 
    GTest::gtest GTest::gtest_main 
    ${LIB_PQXX} ${LIB_PQ}
    ws2_32 mswsock advapi32
)

include(GoogleTest)
gtest_discover_tests(smart_home_tests)
gtest_discover_tests(smart_home_integration_tests)
