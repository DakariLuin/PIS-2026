cmake_minimum_required(VERSION 3.21)

# --- vcpkg toolchain ---
if(DEFINED ENV{VCPKG_ROOT})
    set(CMAKE_TOOLCHAIN_FILE "$ENV{VCPKG_ROOT}/scripts/buildsystems/vcpkg.cmake"
        CACHE STRING "vcpkg toolchain file")
else()
    message(FATAL_ERROR "VCPKG_ROOT is not set. Please set the VCPKG_ROOT environment variable.")
endif()

project(SmartHomeLite VERSION 0.1 LANGUAGES CXX)

# Стандарт C++
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED YES)

# Находим зависимости
find_package(Crow CONFIG REQUIRED)
find_package(GTest CONFIG REQUIRED)

# --- Основной исполняемый файл (REST Сервер) ---
add_executable(smart_home_server 
    src/main.cpp
    src/application/service/reading_service.cpp
)

target_include_directories(smart_home_server PRIVATE src)
target_link_libraries(smart_home_server PRIVATE Crow::Crow)

# ВАЖНО: Сетевые библиотеки для Windows (чтобы Crow линковался)
if(WIN32)
    target_link_libraries(smart_home_server PRIVATE ws2_32 mswsock)
endif()


# --- Юнит-тесты (Google Test) ---
add_executable(smart_home_tests
    tests/reading_service_test.cpp
    tests/test_sensor.cpp           # <--- ДОБАВЛЕНО: Файл с тестами домена (Lab 3)
    src/application/service/reading_service.cpp
)

target_include_directories(smart_home_tests PRIVATE src)
target_link_libraries(smart_home_tests PRIVATE GTest::gtest GTest::gtest_main GTest::gmock)

# Тоже добавляем сети для тестов, если они вызывают логику Crow
if(WIN32)
    target_link_libraries(smart_home_tests PRIVATE ws2_32 mswsock)
endif()

include(GoogleTest)
gtest_discover_tests(smart_home_tests)
